Easy Integration

Get started with our fundamental data API using simple REST endpoints and comprehensive documentation

Python Example

import requests

# Get income statement data
url = "https://api.financialdata.online/fundamentals/AAPL/income-statement"
headers = {"Authorization": "Bearer YOUR_API_KEY"}

response = requests.get(url, headers=headers)
data = response.json()

# Extract key metrics
revenue = data['data']['income_statement']['revenue']
net_income = data['data']['income_statement']['net_income']
eps = data['data']['income_statement']['eps_diluted']

print(f"Revenue: ${revenue:,}")
print(f"Net Income: ${net_income:,}")
print(f"EPS: ${eps}")

JavaScript Example

// Fetch fundamental data
const getFundamentals = async (symbol) => {
  const response = await fetch(
    `https://api.financialdata.online/fundamentals/${symbol}/metrics`,
    {
      headers: {
        'Authorization': 'Bearer YOUR_API_KEY'
      }
    }
  );
  
  const data = await response.json();
  
  // Calculate key ratios
  const pe_ratio = data.data.valuation.pe_ratio;
  const roe = data.data.profitability.return_on_equity;
  const debt_ratio = data.data.leverage.debt_to_equity;
  
  console.log(`P/E Ratio: ${pe_ratio}`);
  console.log(`ROE: ${roe}%`);
  console.log(`Debt/Equity: ${debt_ratio}`);
};

getFundamentals('AAPL');
